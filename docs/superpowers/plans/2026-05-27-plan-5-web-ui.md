# Web UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the complete React + TypeScript web UI for nat-backup: Dashboard, Agents, Jobs, Run History (with live WebSocket logs), Storage Destinations, and Notification Settings. Then embed the production build in the server binary.

**Architecture:** Vite + React + TypeScript + shadcn/ui + TanStack Query. API calls go to `/api/*` (same origin). Auth via JWT stored in localStorage. Real-time run logs via native WebSocket. Production build output to `web/dist/`, embedded in server binary via Go `embed.FS`. Plans 1–4 must be complete first.

**Tech Stack:** React 18, TypeScript, Vite, shadcn/ui, Tailwind CSS v4, TanStack Query v5, React Router v6, Recharts, Lucide React

```typescript
// web/src/lib/auth.ts
import { useState, useCallback } from 'react'

const TOKEN_KEY = 'nat_backup_token'

export function useAuth() {
  const [token, setTokenState] = useState<string | null>(
    () => localStorage.getItem(TOKEN_KEY)
  )

  const setToken = useCallback((t: string | null) => {
    if (t) {
      localStorage.setItem(TOKEN_KEY, t)
    } else {
      localStorage.removeItem(TOKEN_KEY)
    }
    setTokenState(t)
  }, [])

  const logout = useCallback(() => setToken(null), [setToken])

  return { token, setToken, logout }
}

export function getToken(): string | null {
  return localStorage.getItem(TOKEN_KEY)
}
```

```typescript
// web/src/lib/api.ts
import { getToken } from './auth'
import type {
  Agent, BackupJob, BackupRun, StorageDestination,
  NotificationSettings, CreateJobRequest, CreateStorageRequest
} from './types'

class APIError extends Error {
  constructor(public status: number, message: string) {
    super(message)
  }
}

async function request<T>(method: string, path: string, body?: unknown): Promise<T> {
  const token = getToken()
  const headers: Record<string, string> = {
    'Content-Type': 'application/json',
  }
  if (token) {
    headers['Authorization'] = `Bearer ${token}`
  }

  const res = await fetch(path, {
    method,
    headers,
    body: body ? JSON.stringify(body) : undefined,
  })

  if (!res.ok) {
    const err = await res.json().catch(() => ({ error: res.statusText }))
    throw new APIError(res.status, err.error || res.statusText)
  }

  if (res.status === 204) return undefined as T
  return res.json()
}

export const api = {
  // Auth
  login: (username: string, password: string) =>
    request<{ token: string }>('POST', '/api/auth/login', { username, password }),

  // Agents
  listAgents: () => request<Agent[]>('GET', '/api/agents'),
  deleteAgent: (id: string) => request<void>('DELETE', `/api/agents/${id}`),

  // Storage Destinations
  listStorage: () => request<StorageDestination[]>('GET', '/api/storage-destinations'),
  createStorage: (data: CreateStorageRequest) =>
    request<StorageDestination>('POST', '/api/storage-destinations', data),
  updateStorage: (id: string, data: CreateStorageRequest) =>
    request<void>('PUT', `/api/storage-destinations/${id}`, data),
  deleteStorage: (id: string) => request<void>('DELETE', `/api/storage-destinations/${id}`),

  // Jobs
  listJobs: () => request<BackupJob[]>('GET', '/api/jobs'),
  createJob: (data: CreateJobRequest) => request<BackupJob>('POST', '/api/jobs', data),
  updateJob: (id: string, data: CreateJobRequest) =>
    request<void>('PUT', `/api/jobs/${id}`, data),
  deleteJob: (id: string) => request<void>('DELETE', `/api/jobs/${id}`),
  triggerJob: (id: string) => request<BackupRun>('POST', `/api/jobs/${id}/run`),

  // Runs
  listRuns: (params?: { job_id?: string; status?: string }) => {
    const qs = new URLSearchParams(params as Record<string, string>).toString()
    return request<BackupRun[]>('GET', `/api/runs${qs ? '?' + qs : ''}`)
  },

  // Settings
  getNotificationSettings: () =>
    request<NotificationSettings | { configured: false }>('GET', '/api/settings/notifications'),
  updateNotificationSettings: (data: { type: string; config: EmailNotificationConfig }) =>
    request<void>('PUT', '/api/settings/notifications', data),
}

export { APIError }
```

```typescript
import { clsx, type ClassValue } from 'clsx'
import { twMerge } from 'tailwind-merge'

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}

export function formatBytes(bytes: number | null): string {
  if (bytes === null || bytes === 0) return '—'
  const units = ['B', 'KB', 'MB', 'GB', 'TB']
  let i = 0
  let v = bytes
  while (v >= 1024 && i < units.length - 1) {
    v /= 1024
    i++
  }
  return `${v.toFixed(1)} ${units[i]}`
}

export function formatDate(iso: string | null): string {
  if (!iso) return '—'
  return new Intl.DateTimeFormat('en-US', {
    month: 'short', day: 'numeric',
    hour: '2-digit', minute: '2-digit',
  }).format(new Date(iso))
}

export function formatRelative(iso: string | null): string {
  if (!iso) return 'never'
  const diff = Date.now() - new Date(iso).getTime()
  const mins = Math.floor(diff / 60000)
  if (mins < 1) return 'just now'
  if (mins < 60) return `${mins}m ago`
  const hrs = Math.floor(mins / 60)
  if (hrs < 24) return `${hrs}h ago`
  return `${Math.floor(hrs / 24)}d ago`
}

export function cronHumanize(expr: string): string {
  const presets: Record<string, string> = {
    '@daily': 'Every day at midnight',
    '@hourly': 'Every hour',
    '@weekly': 'Every week',
    '@monthly': 'Every month',
    '0 2 * * *': 'Every day at 2:00 AM',
    '0 3 * * *': 'Every day at 3:00 AM',
    '0 0 * * *': 'Every day at midnight',
    '*/5 * * * *': 'Every 5 minutes',
    '0 0 * * 0': 'Every Sunday at midnight',
  }
  return presets[expr] || expr
}
```

```tsx
function useLiveLogs(runID: string | null) {
  const [logs, setLogs] = useState<string[]>([])

  useEffect(() => {
    if (!runID) { setLogs([]); return }

    const proto = window.location.protocol === 'https:' ? 'wss' : 'ws'
    const wsURL = `${proto}://${window.location.host}/api/runs/${runID}/logs/ws`
    const ws = new WebSocket(wsURL)

    ws.onmessage = (e) => {
      try {
        const msg = JSON.parse(e.data)
        setLogs(prev => [...prev, msg.message ?? JSON.stringify(msg)])
      } catch {
        setLogs(prev => [...prev, e.data])
      }
    }

    ws.onerror = () => setLogs(prev => [...prev, '[connection error]'])

    return () => ws.close()
  }, [runID])

  return logs
}

export default function Runs() {
  const [selectedRun, setSelectedRun] = useState<BackupRun | null>(null)
  const [statusFilter, setStatusFilter] = useState('')
  const logsEndRef = useRef<HTMLDivElement>(null)
  const logs = useLiveLogs(selectedRun?.status === 'running' ? selectedRun.id : null)

    queryFn: () => api.listRuns(statusFilter ? { status: statusFilter } : undefined),
    refetchInterval: 5_000,
```

```tsx
import { Dialog, DialogContent, DialogHeader, DialogTitle } from '@/components/ui/dialog'
import { Plus, Trash2 } from 'lucide-react'
import type { StorageDestination, CreateStorageRequest } from '@/lib/types'

type DestType = 'local' | 's3' | 'sftp'

const emptyConfig: Record<DestType, Record<string, string | number | boolean>> = {
  local: { path: '/var/backups/nat-backup' },
  s3: { endpoint: '', bucket: '', region: 'us-east-1', access_key: '', secret_key: '' },
  sftp: { host: '', port: 22, user: '', password: '', path: '/backups' },
}

export default function Storage() {
  const qc = useQueryClient()
  const [formOpen, setFormOpen] = useState(false)
  const [deleting, setDeleting] = useState<StorageDestination | null>(null)
  const [name, setName] = useState('')
  const [type, setType] = useState<DestType>('local')
  const [config, setConfig] = useState<Record<string, unknown>>(emptyConfig.local)

  const { data: storage = [], isLoading } = useQuery({ queryKey: ['storage'], queryFn: api.listStorage })

  const createMutation = useMutation({
    mutationFn: (data: CreateStorageRequest) => api.createStorage(data),
    onSuccess: () => { qc.invalidateQueries({ queryKey: ['storage'] }); setFormOpen(false) },
  })

  const deleteMutation = useMutation({
    mutationFn: (id: string) => api.deleteStorage(id),
    onSuccess: () => { qc.invalidateQueries({ queryKey: ['storage'] }); setDeleting(null) },
  })

  function handleTypeChange(t: DestType) {
    setType(t)
    setConfig(emptyConfig[t])
  }

  function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    createMutation.mutate({ name, type, config })
  }

  if (isLoading) return <div className="text-muted-foreground">Loading...</div>

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <h2 className="text-2xl font-bold">Storage Destinations</h2>
        <Button onClick={() => setFormOpen(true)} className="gap-2">
          <Plus className="h-4 w-4" /> Add Destination
        </Button>
      </div>

      <Card>
        <CardContent className="pt-4">
          {storage.length === 0 && (
            <p className="text-center py-8 text-muted-foreground">No storage destinations configured.</p>
          )}
          <div className="space-y-3">
            {storage.map(dest => (
              <div key={dest.id} className="flex items-center justify-between p-4 border rounded-lg">
                <div>
                  <div className="font-medium">{dest.name}</div>
                  <div className="text-sm text-muted-foreground capitalize">{dest.type}</div>
                </div>
                <Button variant="ghost" size="icon" onClick={() => setDeleting(dest)}>
                  <Trash2 className="h-4 w-4 text-destructive" />
                </Button>
              </div>
            ))}
          </div>
        </CardContent>
      </Card>

      <Dialog open={formOpen} onOpenChange={(o) => !o && setFormOpen(false)}>
        <DialogContent className="max-w-lg">
          <DialogHeader><DialogTitle>Add Storage Destination</DialogTitle></DialogHeader>
          <form onSubmit={handleSubmit} className="space-y-4">
            <div className="space-y-2">
              <Label>Name</Label>
              <Input value={name} onChange={e => setName(e.target.value)} placeholder="My S3 Bucket" required />
            </div>
            <div className="space-y-2">
              <Label>Type</Label>
              <Select value={type} onValueChange={(v) => handleTypeChange(v as DestType)}>
                <SelectTrigger><SelectValue /></SelectTrigger>
                <SelectContent>
                  <SelectItem value="local">Local (server disk)</SelectItem>
                  <SelectItem value="s3">S3-Compatible</SelectItem>
                  <SelectItem value="sftp">SFTP</SelectItem>
                </SelectContent>
              </Select>
            </div>

            {/* Dynamic config fields */}
            {Object.entries(config).map(([key, val]) => (
              <div key={key} className="space-y-2">
                <Label className="capitalize">{key.replace(/_/g, ' ')}</Label>
                <Input
                  type={key.includes('password') || key.includes('key') ? 'password' : 'text'}
                  value={String(val)}
                  onChange={e => setConfig(c => ({ ...c, [key]: e.target.value }))}
                />
              </div>
            ))}

            <div className="flex justify-end gap-2">
              <Button type="button" variant="outline" onClick={() => setFormOpen(false)}>Cancel</Button>
              <Button type="submit" disabled={createMutation.isPending}>Add</Button>
            </div>
          </form>
        </DialogContent>
      </Dialog>

      <ConfirmDialog
        open={!!deleting}
        title="Delete Storage Destination"
        description={`Delete "${deleting?.name}"? Jobs using this destination will fail.`}
        onConfirm={() => deleting && deleteMutation.mutate(deleting.id)}
        onCancel={() => setDeleting(null)}
        loading={deleteMutation.isPending}
      />
    </div>
  )
}
```

```tsx
import { useState, useEffect } from 'react'
import { useQuery, useMutation } from '@tanstack/react-query'
import { api } from '@/lib/api'

import type { EmailNotificationConfig } from '@/lib/types'

const defaultEmailCfg: EmailNotificationConfig = {
  smtp_host: '', smtp_port: 587,
  from: '', to: [],
  tls: true, username: '', password: '',
}

export default function Settings() {
  const [emailCfg, setEmailCfg] = useState<EmailNotificationConfig>(defaultEmailCfg)
  const [toInput, setToInput] = useState('')
  const [saved, setSaved] = useState(false)

  const { data: settings } = useQuery({
    queryKey: ['notification-settings'],
    queryFn: api.getNotificationSettings,
  })

  useEffect(() => {
    if (settings && 'config' in settings) {
      setEmailCfg(settings.config)
      setToInput(settings.config.to.join(', '))
    }
  }, [settings])

  const saveMutation = useMutation({
    mutationFn: (data: { type: string; config: EmailNotificationConfig }) =>
      api.updateNotificationSettings(data),
    onSuccess: () => { setSaved(true); setTimeout(() => setSaved(false), 3000) },
  })

  function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    const cfg = {
      ...emailCfg,
      to: toInput.split(',').map(s => s.trim()).filter(Boolean),
    }
    saveMutation.mutate({ type: 'email', config: cfg })
  }

  return (
    <div className="space-y-6 max-w-2xl">
      <h2 className="text-2xl font-bold">Settings</h2>

      <Card>
        <CardHeader><CardTitle>Email Notifications</CardTitle></CardHeader>
        <CardContent>
          <form onSubmit={handleSubmit} className="space-y-4">
            <div className="grid grid-cols-2 gap-4">
              <div className="space-y-2">
                <Label>SMTP Host</Label>
                <Input value={emailCfg.smtp_host}
                  onChange={e => setEmailCfg(c => ({ ...c, smtp_host: e.target.value }))}
                  placeholder="smtp.gmail.com" />
              </div>
              <div className="space-y-2">
                <Label>SMTP Port</Label>

                  onChange={e => setEmailCfg(c => ({ ...c, smtp_port: Number(e.target.value) }))} />
              </div>
            </div>

            <div className="grid grid-cols-2 gap-4">
              <div className="space-y-2">
                <Label>Username</Label>
                <Input value={emailCfg.username ?? ''}
                  onChange={e => setEmailCfg(c => ({ ...c, username: e.target.value }))} />
              </div>
              <div className="space-y-2">
                <Label>Password</Label>
                <Input type="password" value={emailCfg.password ?? ''}
                  onChange={e => setEmailCfg(c => ({ ...c, password: e.target.value }))} />
              </div>
            </div>

            <div className="space-y-2">
              <Label>From address</Label>
              <Input value={emailCfg.from}
                onChange={e => setEmailCfg(c => ({ ...c, from: e.target.value }))}
                placeholder="backup@example.com" />
            </div>

            <div className="space-y-2">
              <Label>Recipients (comma-separated)</Label>

            </div>

                <span className="text-sm text-destructive">Failed to save.</span>
              )}
            </div>
          </form>
        </CardContent>
      </Card>
    </div>
  )
}
```

```go
// Add to internal/api/server.go

import (
	"io/fs"
	"net/http"

	apiStatic "github.com/felipendelicia/nat-backup/internal/api/static"
)

// NewServerWithStatic creates a server that serves embedded React UI.
func NewServerWithStatic(db *sqlx.DB, cfg config.Config) http.Handler {
	hub := ws.NewHub(db)
	go hub.Run()

	s := &Server{
		db:        db,
		cfg:       cfg,
		hub:       hub,
		wsHandler: ws.NewAgentHandler(hub),
	}
	s.router = s.buildRouterWithStatic()
	return s
}

func (s *Server) buildRouterWithStatic() chi.Router {
	r := s.buildRouter()

	// Serve embedded React UI for all non-API routes
	distFS, err := fs.Sub(apiStatic.FS, "dist")
	if err != nil {
		// No embedded UI — skip static serving (dev mode)
		return r
	}

	fileServer := http.FileServer(http.FS(distFS))

	r.Get("/*", func(w http.ResponseWriter, r *http.Request) {
		// Try to serve static file; fall back to index.html for SPA routing
		_, err := distFS.Open(r.URL.Path[1:])
		if err != nil {
			// Serve index.html for all unknown paths (client-side routing)
			r2 := *r
			r2.URL.Path = "/"
			fileServer.ServeHTTP(w, &r2)
			return
		}
		fileServer.ServeHTTP(w, r)
	})

	return r
}
```
